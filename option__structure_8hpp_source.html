585 {

605
606 // Check if any bad character is input.
607 if( istr.fail() )
608 Terminate("AddScalarOption", __FILE__, __LINE__,
609 "Wrong character input inside expected parameter type!");
610
611 // Go back to previous line location.
612 if( ResetLine ) inputFile.seekg(CurrentLine);

798 std::string keyword,
799 unsigned short nExpected,
800 unsigned short nCondition1 = 0,
801 unsigned short nCondition2 = 0,
802 unsigned short nCondition3 = 0
803)
804 /*
805 * Function that pds the remaining entries in a given vector of templated value.
806 */
807{
808 // Size of current data vector.
809 unsigned short nData = data.size();
810
811 // Check first condition.
812 if( nData != nExpected ){
813
814 // Data flag.
815 bool PadData = false;
816 // Check if there need be any modification, if so flag the data.
817 if( nData == nCondition1 ) PadData = true;
818 if( nData == nCondition2 ) PadData = true;
819 if( nData == nCondition3 ) PadData = true;
820
821 // Pad data accordingly.
822 if( PadData ) for(unsigned short i=nData; i<nExpected; i++) data.push_back( data[i-1] );
823 }
824
825 // Consistency check.
826 if( data.size() != nExpected ){
827 std::string message = "Keyword: ";
828 message += keyword;
829 message += " must be of size: ";
